# 二叉树的遍历

二叉树有三种遍历方式：

  • 先序遍历：根-左-右
  • 中序遍历：左-根-右
  • 后序遍历：左-右-根 这三种遍历方式是根据根结点的输出顺序的不同来命名的。

例如以该二叉树为例

let tree = {
  val: 6,
  left: {
    val: 4,
    left: {
      val: 3
    },
    right: {
      val: 5
    }
  },
  right: {
    val: 8,
    left: {
      val: 7
    },
    right: {
      val: 9
    }
  }
}
/*
   6
 4   8
3 5 7 9
*/

该二叉树的三种遍历顺序输出顺序分别为

  • 先序遍历：6,4,3,5,8,7,9
  • 中序遍历：3,4,5,6,7,8,9
  • 后序遍历：3,5,4,7,9,8,6

三种遍历方式都有两种实现方式，递归实现和非递归实现，以下代码中，D代表根结点，L代表左节点，R代表右节点

# 递归实现

// 先序遍历
function DLR(node) {
  if (!node) return null
  console.log(node.val)
  DLR(node.left)
  DLR(node.right)
}

// 中序遍历
function LDR(node) {
  if (!node) return null
  LDR(node.left)
  console.log(node.val)
  LDR(node.right)
}

// 后序遍历
function LRD(node) {
  if (!node) return null
  LRD(node.left)
  LRD(node.right)
  console.log(node.val)
}

递归实现非常简单，因为该实现方式更符合我们的思维逻辑，但是我们知道，普通的递归性能都比较慢，所以我们也需要掌握非递归的实现。

# 非递归实现

非递归实现可以看作是通过栈的操作来模拟递归的操作

// 先序遍历
function DLR(node) {
  let arr = [], res = []
  if (node) {
    arr.push(node)
  }
  while (arr.length) {
    let temp = arr.pop()
    res.push(temp.val)
    // 因为用栈模拟，所以左边的节点最后推入，pop时得到的就是左边的节点
    if (temp.right) {
      arr.push(temp.right)
    }
    if (temp.left) {
      arr.push(temp.left)
    }
  }
  return res
}

// 中序遍历
function LDR(node) {
  let arr = [], res = []
  while(true) {
    // 先把根结点和左子树全部推入栈中
    while(node) {
      arr.push(node)
      node = node.left
    }
    
    // 当栈为空时，说明树已遍历结束
    if (arr.length === 0) break;

    // 得到左子树节点，推入右子树
    let temp = arr.pop()
    res.push(temp.val)
    node = temp.right
  }

  return res
}

// 后序遍历
function LRD(node) {
  let arr=[], res=[]
  arr.push(node)

  while (arr.length) {
    var temp = arr.pop()
    res.push(temp.val)
    if (temp.left){
      arr.push(temp.left)
    }
    if (temp.right){
      arr.push(temp.right)
    }
  }

  return res.reverse();
}
