# 从尾到头打印链表

输入一个链表，按链表从尾到头的顺序返回一个ArrayList。

在js内实现从尾到头打印链表，只需要从头遍历链表，每次都将值插入另一个数组的头部即可。

function printListFromTailToHead (head) {
  let res = [], node = head
  while (node !== null) {
    // unshift方法可以将值插入数组的头部
    res.unshift(node.val)
    node = node.next
  }
  return res
}
